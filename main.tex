  (aux1) -- coordinate (a)

  ++(-45:5.1) coordinate (b);

  (aux3) --
  (a) --
  (aux1);

  (b) --
  ++(225:2.2) --
  ++(-45:2.2);
\end{scope}

  (aux4) --
  ++(225:0.8) --
  ++(-45:0.8);

\draw[shorten <= -10pt]
  (aux2) --
  ++(225:3) coordinate[pos=0.45] (c) --
  ++(-45:3.1);
\draw
  (aux2) --
  (c) --
  ++(135:2.5) --
  ++(45:2.5) --
  ++(-45:2.5) coordinate[pos=0.3] (d);   
\draw 
  (d) -- +(45:1);
\end{scope}
\end{tikzpicture}%
